22

            CONTINUE  
        END  IF  
         
        //  Step  3:  Base  Criteria  Evaluation  
        IF  s_QPI  >=  Grant.MinimumQPI  AND  s_Income  <=  Grant.MaximumIncome  THEN  
            //  Step  4:  Degree  Program  Matching

                //  Step  5:  Policy  Contradiction  &  Exclusion  Resolution

                END  IF    
 
23

                EligibleGrants.APPEND(Grant)  
            END  IF  
        END  IF
